TAD IrAlBaño{
    obs bañoOcupado: bool
    obs alumnos: Seq<Alumno>
    obs cola: Cola<Alumno> 
    
    proc arrancaElParcial():IrAlBaño
        asegura: res.bañoOcupado = False
        asegura: forall i:int :: 0<=i<|res.alumnos| ==>L res.alumnos[i] = 0 //todos arrancan yendo 0 veces
        asegura: res.cola = {}
    
    proc quiereIrAlBaño(inout a:IrAlBaño, in alumno:ZxZ) //se suma a la fila
        requiere: alumno in a.alumnos
        asegura: alumno < 3 ==> b.cola.encolar
    
    proc avanzarAlBaño(inout a:IrAlBaño)
        requiere: a.bañoOcupado = False
        asegura: a.cola = old(a).cola.desencolar(a.cola) //hace falta pasarle el a.cola????
        asegura: a.bañoOcupado = True

    proc estaEnLaCola?(in a:IrAlBaño, in alumno: ZxZ):bool
        requiere: alumno in a.alumnos && alumno[1] < 3
        asegura: res = True <==> alumno in a.cola
}
